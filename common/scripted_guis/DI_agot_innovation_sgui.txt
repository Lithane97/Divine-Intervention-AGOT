### Check innovation, add innovation ###
# outriders
DI_culture_has_innovation_outriders = {
	scope = culture

	is_valid = {
		has_innovation = innovation_outriders
	}
}

DI_add_innovation_outriders = {
	scope = character

	effect = {
		var:DI_innovation_selected_culture = {
			add_innovation = innovation_outriders
		}
	}
}
