### ARMY SPAWNING EFFECTS ###
### Initialize
DI_army_spawner_resources_sgui = {
	scope = character
	effect = {
		if = {
			limit = {
				NOT = {
					has_global_variable = DI_army_spawner_increment_value
				}
			}
			set_global_variable = {
				name = DI_army_spawner_increment_value
				value = 500
			}
		}
	}
}

DI_army_spawner_add_increment = {
	is_valid = {
		global_var:DI_army_spawner_increment_value < 100000
	}
	effect = {
		if = {
			limit = {
				global_var:DI_army_spawner_increment_value >= 50000
			}
			change_global_variable = {
				name = DI_army_spawner_increment_value
				add = 10000
			}
		}
		else_if = {
			limit = {
				AND = {
					global_var:DI_army_spawner_increment_value < 50000
					global_var:DI_army_spawner_increment_value >= 10000
				}
			}
			change_global_variable = {
				name = DI_army_spawner_increment_value
				add = 5000
			}
		}
		else_if = {
			limit = {
				AND = {
					global_var:DI_army_spawner_increment_value < 10000
					global_var:DI_army_spawner_increment_value >= 5000
				}
			}
			change_global_variable = {
				name = DI_army_spawner_increment_value
				add = 1000
			}
		}
		else = {
			change_global_variable = {
				name = DI_army_spawner_increment_value
				add = 500
			}
		}
	}
}

	scope = character
	effect = {
		spawn_army = {
			men_at_arms = {
				type = light_footmen

			name = light_footmen
		}
	}
